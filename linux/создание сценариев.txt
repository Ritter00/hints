-----------создать сценарий запуска python в Linux----------------
создать файл(напр. script)
nano script
# в нем записать и сохранить

#!/usr/bin/env python3 или #!/usr/bin/python3
# программа env определяет местоположение интерпретатора Python согласно настройкам пути поиска в системе
print('The Bright Side ' + 'of Life...')

chmod +x script		# разрешить уму исполнение
./script		# запустить сценарий
